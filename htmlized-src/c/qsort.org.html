#include <stdio.h>
#include <stdlib.h>
#include <malloc.h>
#include <string.h>
//统计文件中字符数,单词数,行数
int comp(int *a ,int *b ){
  return *a<*b;
}

int main(int argc, char *argv[])
{
  int i,s[] ={ 1,3 ,12,5 };
  /* 声明一个函数指针,它指向一个函数 */
  int (*c)()=comp;

  /* qsort() 函数,第一个是待排序的数组,
     参数2,是数组的长度,
     参数3是每个元素的长度
     *参数4是一个函数指针,这个函数接受两个参数,
     此函数用于比较
  */
  qsort(s,4,sizeof(int),c);

  for(i=0;i<4;i++){
    printf ("%d,",s[i]);
  }
  return 0;
}
